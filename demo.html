StarSpark.js!

Javascript library for crating sparkling text or effects

This is very early in development

Examples

let element = document.querySelector('#flash')
element.addEventListener('click',() => {
new StarSpark(element, {
    'totalDuration': 0.25,
    'sparkDuration': 0.25,
    'sparkSize': 1.5,
    'sparkHTML':'✦',
    });
});
                
let element = document.querySelector('#glitter')
new StarSpark(element, {
    'totalDuration': 5,
    'sparkDuration': 0.5,
});
                
let element = document.querySelector('#joke')
new StarSpark(element, {
    'overflow': 30,
    'totalDuration': 5,
    'sparkDuration': 0.25,
    'sparksPerCycle':2,
    'sparkHTML':'HA',
});
